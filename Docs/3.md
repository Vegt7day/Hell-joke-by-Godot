2.2 状态机

scripts/core/states/state_machine.gd:
extends Node
# 状态机

class_name StateMachine

# 信号
signal state_changed(old_state: String, new_state: String)
signal transition_started(from_state: String, to_state: String)
signal transition_completed(to_state: String)

# 属性
@export var initial_state: String = "idle"
@export var debug_mode: bool = false

# 状态
var states: Dictionary = {}
var current_state: State
var previous_state: String = ""
var state_history: Array[String] = []
var actor: Node
var is_transitioning: bool = false

func _ready():
	actor = get_parent()
	setup_states()
	start()

func setup_states():
	"""初始化所有子状态节点"""
	for child in get_children():
		if child is State:
			child.setup(self, actor)
			states[child.name.to_lower()] = child
			child.transition_requested.connect(_on_state_transition_requested)
	
	print("状态机初始化完成，状态数量: %d" % states.size())

func start():
	"""启动状态机"""
	if initial_state in states:
		transition_to(initial_state)
	else:
		push_error("初始状态不存在: %s" % initial_state)

func transition_to(state_name: String, data: Dictionary = {}):
	"""转换到指定状态"""
	if is_transitioning:
		print("警告：正在状态转换中，请等待")
		return
	
	var new_state = states.get(state_name.to_lower())
	if not new_state:
		print("错误：状态不存在: %s" % state_name)
		return
	
	if current_state and current_state.name == new_state.name:
		if debug_mode:
			print("已在目标状态: %s" % state_name)
		return
	
	is_transitioning = true
	
	# 记录状态历史
	if current_state:
		state_history.append(current_state.name)
		if state_history.size() > 10:  # 限制历史长度
			state_history.pop_front()
	
	var old_state_name = current_state.name if current_state else "none"
	
	emit_signal("transition_started", old_state_name, state_name)
	
	if debug_mode:
		print("状态转换: %s -> %s" % [old_state_name, state_name])
	
	# 退出当前状态
	if current_state:
		current_state.exit()
	
	# 更新状态
	previous_state = old_state_name
	current_state = new_state
	
	# 进入新状态
	current_state.enter(data)
	
	emit_signal("state_changed", previous_state, state_name)
	emit_signal("transition_completed", state_name)
	
	is_transitioning = false

func get_state(state_name: String) -> State:
	"""获取状态实例"""
	return states.get(state_name.to_lower())

func get_current_state_name() -> String:
	"""获取当前状态名称"""
	return current_state.name if current_state else "none"

func is_in_state(state_name: String) -> bool:
	"""检查是否在指定状态"""
	return current_state and current_state.name.to_lower() == state_name.to_lower()

func can_transition_to(state_name: String) -> bool:
	"""检查是否可以转换到指定状态"""
	return state_name in states and not is_transitioning

func go_back():
	"""返回到上一个状态"""
	if state_history.size() > 0:
		var previous = state_history.pop_back()
		transition_to(previous)
		return true
	return false

func _on_state_transition_requested(state_name: String, data: Dictionary = {}):
	"""处理状态转换请求"""
	transition_to(state_name, data)

func _process(delta: float):
	if current_state:
		current_state.process(delta)

func _physics_process(delta: float):
	if current_state:
		current_state.physics_process(delta)

func _input(event: InputEvent):
	if current_state:
		current_state.input(event)

func _unhandled_input(event: InputEvent):
	if current_state:
		current_state.unhandled_input(event)

func handle_input():
	"""处理输入"""
	if not can_move:
		return
	
	# 获取输入方向
	input_direction.x = Input.get_axis("move_left", "move_right")
	
	# 更新面向方向
	if input_direction.x != 0:
		facing_direction = sign(input_direction.x)
		if sprite:
			sprite.scale.x = abs(sprite.scale.x) * facing_direction
	
	# 处理Q键：切换肢体
	if Input.is_action_just_pressed("attack_q"):
		switch_limb()
	
	# 处理J键：使用当前肢体
	if Input.is_action_just_pressed("attack_j"):
		use_active_limb()
	
	# 处理R键：回收
	if Input.is_action_just_pressed("reload_r"):
		recover_limbs()

func handle_movement(delta: float):
	"""处理移动"""
	var target_velocity = Vector2.ZERO
	
	# 水平移动
	if input_direction.x != 0:
		target_velocity.x = input_direction.x * move_speed
		velocity.x = move_toward(velocity.x, target_velocity.x, acceleration * delta)
	else:
		# 减速
		velocity.x = move_toward(velocity.x, 0, deceleration * delta)
	
	# 垂直速度（重力处理在apply_gravity中）
	pass

func apply_gravity(delta: float):
	"""应用重力"""
	if not is_on_floor():
		velocity.y += gravity * delta
		velocity.y = min(velocity.y, max_fall_speed)

func update_timers(delta: float):
	"""更新计时器"""
	# Coyote时间
	if is_on_floor():
		coyote_timer = coyote_time
	else:
		coyote_timer = max(coyote_timer - delta, 0.0)
	
	# 跳跃缓冲
	if Input.is_action_just_pressed("jump"):
		jump_buffer_timer = jump_buffer_time
	else:
		jump_buffer_timer = max(jump_buffer_timer - delta, 0.0)

func handle_jump(delta: float):
	"""处理跳跃"""
	# 重置跳跃计数
	if is_on_floor():
		jump_count = 0
		can_double_jump = true
	
	# 处理跳跃输入
	if jump_buffer_timer > 0:
		if is_on_floor() or coyote_timer > 0:
			perform_jump()
		elif can_double_jump and jump_count < 1:
			perform_double_jump()

func perform_jump():
	"""执行跳跃"""
	velocity.y = -jump_force
	jump_count += 1
	jump_buffer_timer = 0.0
	
	# 播放跳跃音效
	play_jump_sound()
	
	print("执行跳跃")

func perform_double_jump():
	"""执行二段跳"""
	velocity.y = -double_jump_force
	jump_count += 1
	can_double_jump = false
	jump_buffer_timer = 0.0
	
	# 播放二段跳音效
	play_double_jump_sound()
	
	print("执行二段跳")

func switch_limb():
	"""切换肢体"""
	print("切换肢体")
	
	var limb_keys = limbs.keys()
	var current_index = limb_keys.find(active_limb)
	
	if current_index == -1:
		active_limb = limb_keys[0]
	else:
		# 查找下一个激活的肢体
		var next_index = (current_index + 1) % limb_keys.size()
		for i in range(limb_keys.size()):
			var limb_name = limb_keys[(next_index + i) % limb_keys.size()]
			if limbs[limb_name]["active"]:
				active_limb = limb_name
				limb_history.append(limb_name)
				if limb_history.size() > 5:
					limb_history.pop_front()
				break
	
	print("切换到肢体: %s" % active_limb)
	
	# 更新UI
	if EventBus.instance:
		EventBus.instance.player_form_changed.emit(limb_history[-2] if limb_history.size() > 1 else "none", active_limb)

func use_active_limb():
	"""使用当前活动肢体"""
	if not limbs[active_limb]["active"]:
		print("肢体未激活: %s" % active_limb)
		return
	
	print("使用肢体: %s" % active_limb)
	
	match active_limb:
		"pen":
			use_pen()
		"left_shoe", "right_shoe":
			use_shoe(active_limb)
		"book":
			use_book()
		_:
			print("未知肢体: %s" % active_limb)

func recover_limbs():
	"""回收肢体"""
	print("回收肢体")
	
	# 回收墨
	recover_ink()
	
	# 回收鞋
	recover_shoes()
	
	# 恢复耐久度
	for limb in limbs:
		if limbs[limb]["durability"] is float:
			limbs[limb]["durability"] = limbs[limb]["max_durability"]
	
	# 播放回收音效
	play_recover_sound()

func use_pen():
	"""使用笔"""
	if ink <= 0:
		print("墨水不足")
		return
	
	# 消耗墨水
	var ink_cost = 10.0
	if ink >= ink_cost:
		ink -= ink_cost
		ink_changed.emit(ink + ink_cost, ink)
		
		# 发射墨
		emit_ink_projectile()
		
		# 笔颜色变淡
		var current_color = limbs["pen"]["color"]
		var new_color = current_color * Color(1.0, 1.0, 1.0, 0.8)
		limbs["pen"]["color"] = new_color
		
		# 减少耐久
		limbs["pen"]["durability"] = max(0, limbs["pen"]["durability"] - 5)
		
		print("发射墨，剩余墨水: %f" % ink)
		
		# 播放音效
		play_ink_sound()
	else:
		print("墨水不足")

func recover_ink():
	"""回收墨"""
	var ink_recover = 20.0
	ink = min(max_ink, ink + ink_recover)
	ink_changed.emit(ink - ink_recover, ink)

	print("回收鞋子")

func apply_speed_boost():
	"""应用速度提升"""
	var original_speed = move_speed
	move_speed *= 1.5
	
	# 5秒后恢复
	await get_tree().create_timer(5.0).timeout
	move_speed = original_speed
	
	print("速度提升效果结束")

func start_bleed_effect():
	"""开始流血效果"""
	print("开始流血效果")
	
	# 每2秒扣1%最大生命值
	var bleed_timer = 2.0
	while bleed_timer > 0 and is_alive:
		await get_tree().create_timer(2.0).timeout
		if is_alive:
			var bleed_damage = max_health * 0.01
			take_damage(bleed_damage, "bleed")
		bleed_timer -= 2.0

func take_damage(amount: float, source: String = "unknown"):
	"""受到伤害"""
	if not is_alive:
		return
	
	var old_health = health
	health = max(0, health - amount)
	health_changed.emit(old_health, health)
	
	print("受到伤害: %f, 来源: %s, 剩余生命: %f" % [amount, source, health])
	
	if EventBus.instance:
		EventBus.instance.player_hurt.emit(amount, health)
	
	if health <= 0:
		die(source)

func heal(amount: float):
	"""治疗"""
	var old_health = health
	health = min(max_health, health + amount)
	health_changed.emit(old_health, health)
	
	print("治疗: %f, 当前生命: %f" % [amount, health])
	
	if EventBus.instance:
		EventBus.instance.player_healed.emit(amount, health)

func die(cause: String = "unknown"):
	"""死亡"""
	if not is_alive:
		return
	
	is_alive = false
	can_move = false
	
	print("玩家死亡，原因: %s" % cause)
	
	if EventBus.instance:
		EventBus.instance.player_died.emit(cause)
	
	player_died.emit(cause)
	
	# 播放死亡动画
	if animation_player:
		animation_player.play("death")
	
	# 等待后重生
	await get_tree().create_timer(2.0).timeout
	respawn()

func apply_student_speed_boost():
	"""学生形态应用速度提升"""
	print("应用学生速度提升")
	
	# 保存原始速度
	if speed_boost_timer <= 0:
		original_move_speed = move_speed
	
	# 应用速度提升
	move_speed = original_move_speed * speed_boost_multiplier
	speed_boost_timer = speed_boost_duration
	
	# 创建计时器
	if speed_boost_timer > 0:
		await get_tree().create_timer(speed_boost_duration).timeout
		if speed_boost_timer > 0:  # 检查是否被其他效果重置
			end_speed_boost()

	"""结束速度提升"""
	print("结束速度提升")
	
	move_speed = original_move_speed
	speed_boost_timer = 0.0

func start_student_bleed_effect():
	"""学生形态开始流血效果"""
	if is_bleeding:
		# 如果已经在流血，重置计时器
		bleed_timer = bleed_duration
		return
	
	print("开始学生流血效果")
	
	is_bleeding = true
	bleed_timer = bleed_duration
	
	# 开始流血协程
	start_bleeding_coroutine()

func start_bleeding_coroutine():
	"""开始流血协程"""
	while bleed_timer > 0 and is_alive:
		await get_tree().create_timer(1.0).timeout
		if is_alive and is_bleeding:
			# 每秒扣血
			var bleed_damage = bleed_damage_per_second
			take_damage(bleed_damage, "bleed")
			
			# 减少计时器
			bleed_timer -= 1.0
			
			print("流血伤害: %f, 剩余时间: %f" % [bleed_damage, bleed_timer])
	
	# 流血结束
	end_bleeding_effect()

func end_bleeding_effect():

	is_bleeding = false
	bleed_timer = 0.0

func _init():
	name = "idle"

func enter(data: Dictionary = {}):
	super.enter(data)
	
	if actor and actor.sprite and actor.animation_player:
		actor.animation_player.play("idle")
	
	print("进入空闲状态")

func physics_process(delta: float):
	super.physics_process(delta)
	
	# 检查是否应该转换到移动状态
	if actor.input_direction.x != 0:
		transition_to("move")
	
	# 检查是否在地面
	if not actor.is_on_floor():
		transition_to("fall")

func input(event: InputEvent):
	super.input(event)
	
	# 处理跳跃输入
	if event.is_action_pressed("jump"):
		if actor.is_on_floor() or actor.coyote_timer > 0:
			transition_to("jump")
	
	# 处理攻击输入
	if event.is_action_pressed("attack_q"):
		transition_to("attack", {"attack_type": "q"})
	elif event.is_action_pressed("attack_j"):
		transition_to("attack", {"attack_type": "j"})
	elif event.is_action_pressed("reload_r"):
		transition_to("recover")

func exit():
	super.exit()
	print("退出空闲状态")

class_name AttackState

var attack_type: String = "q"
var attack_timer: float = 0.3  # 攻击持续时间

func _init():
	name = "attack"

func enter(data: Dictionary = {}):
	super.enter(data)
	
	attack_type = data.get("attack_type", "q")
	
	if actor and actor.sprite and actor.animation_player:
		actor.animation_player.play("attack_" + attack_type)
	
	# 执行攻击
	perform_attack()
	
	print("进入攻击状态: %s" % attack_type)

func physics_process(delta: float):
	super.physics_process(delta)
	
	# 攻击计时
	attack_timer = max(0, attack_timer - delta)
	
	if attack_timer <= 0:
		# 返回到适当的状态
		if not actor.is_on_floor():
			if actor.velocity.y > 0:
				transition_to("fall")
			else:
				transition_to("jump")
		elif actor.input_direction.x != 0:
			transition_to("move")
		else:
			transition_to("idle")

func perform_attack():
	"""执行攻击"""
	match attack_type:
		"q":
			# Q键攻击：切换肢体
			actor.switch_limb()
		"j":
			# J键攻击：使用当前肢体
			actor.use_active_limb()
		_:
			print("未知攻击类型: %s" % attack_type)

func exit():
	super.exit()
	print("退出攻击状态")

import os
from PIL import Image, ImageDraw

def create_placeholder_texture(name: str, width: int, height: int, color: tuple):
    """创建占位纹理"""
    
    # 创建图像
    img = Image.new('RGBA', (width, height), color)
    draw = ImageDraw.Draw(img)
    
    # 添加边框
    draw.rectangle([(0, 0), (width-1, height-1)], outline=(255, 255, 255, 255))
    
    # 添加文本
    from PIL import ImageFont
    try:
        font = ImageFont.truetype("arial.ttf", 12)
    except:
        font = ImageFont.load_default()
    
    draw.text((10, 10), name, fill=(255, 255, 255, 255), font=font)
    
    # 保存
    output_dir = "assets/graphics/characters/student/"
    os.makedirs(output_dir, exist_ok=True)
    
    img.save(os.path.join(output_dir, f"{name}.png"))
    print(f"创建占位纹理: {name}.png")
