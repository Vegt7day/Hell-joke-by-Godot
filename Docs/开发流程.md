scenes/system/game_manager.gd:
extends Node

class_name GameManager

# 单例实例
static var instance: GameManager

# 游戏状态
enum GameState { MAIN_MENU, PLAYING, PAUSED, CUTSCENE, GAME_OVER }
var current_state: GameState = GameState.MAIN_MENU

# 玩家数据
var player_data: Dictionary = {
	"current_level": 1,
	"health": 100,
	"current_form": "student",
	"unlocked_forms": ["student"],
	"inventory": [],
	"choices": {}
}

func _init():
	if instance == null:
		instance = self
	else:
		queue_free()

func _ready():
	set_process_mode(Node.PROCESS_MODE_ALWAYS)
	
func change_state(new_state: GameState):
	var old_state = current_state
	current_state = new_state
	
	match new_state:
		GameState.PLAYING:
			Input.set_mouse_mode(Input.MOUSE_MODE_HIDDEN)
		GameState.PAUSED:
			Input.set_mouse_mode(Input.MOUSE_MODE_VISIBLE)
		GameState.MAIN_MENU:
			Input.set_mouse_mode(Input.MOUSE_MODE_VISIBLE)
	
	EventBus.emit_signal("game_state_changed", old_state, new_state)

func load_game(slot: int = 0):
	var save_data = SaveManager.load_game(slot)
	if save_data:
		player_data = save_data
		return true
	return false

func save_game(slot: int = 0):
	return SaveManager.save_game(slot, player_data)

# 信号
signal health_changed(old_health: int, new_health: int)

# 属性
@export var max_health: int = 100
@export var move_speed: float = 200.0
@export var jump_force: float = 400.0
@export var gravity: float = 980.0

var health: int = max_health
var current_form: String = "student"
var is_alive: bool = true
var can_move: bool = true

# 子节点
@onready var sprite: Sprite2D = $Sprite2D
@onready var animation_player: AnimationPlayer = $AnimationPlayer
@onready var collision_shape: CollisionShape2D = $CollisionShape2D

func _ready():
	# 初始化
	health = max_health
	setup_form()

func _physics_process(delta: float):
	if not is_alive or not can_move:
		return
	
	# 应用重力
	if not is_on_floor():
		velocity.y += gravity * delta
	
	# 移动
	var input_direction = Input.get_axis("move_left", "move_right")
	velocity.x = input_direction * move_speed
	
	# 跳跃
	if Input.is_action_just_pressed("jump") and is_on_floor():
		velocity.y = -jump_force
	
	move_and_slide()
	update_animation()

func take_damage(amount: int):
	if not is_alive:
		return
	
	var old_health = health
	health = max(0, health - amount)
	emit_signal("health_changed", old_health, health)
	
	if health <= 0:
		die()

func heal(amount: int):
	var old_health = health
	health = min(max_health, health + amount)
	emit_signal("health_changed", old_health, health)

func die():
	is_alive = false
	can_move = false
	emit_signal("died")
	
	# 死亡动画
	animation_player.play("death")
	
	# 重生
	await get_tree().create_timer(1.0).timeout
	respawn()

func respawn():
	health = max_health
	is_alive = true
	can_move = true
	global_position = Vector2.ZERO  # 应设为检查点位置

4. 测试驱动开发流程

创建测试脚本:
# test/test_player.gd
extends Node
class_name PlayerTest

func test_player_movement():
	var player = preload("res://scenes/actors/player/student/student.tscn").instantiate()
	add_child(player)
	
	# 测试移动
	Input.action_press("move_right")
	await get_tree().create_timer(0.1).timeout
	assert(player.velocity.x > 0, "Player should move right")
	Input.action_release("move_right")
	
	# 测试跳跃
	var initial_y = player.global_position.y
	Input.action_press("jump")
	await get_tree().create_timer(0.1).timeout
	assert(player.global_position.y < initial_y, "Player should jump")
	
	player.queue_free()

func test_damage_system():
	var player = preload("res://scenes/actors/player/student/student.tscn").instantiate()
	add_child(player)
	
	var initial_health = player.health
	player.take_damage(20)
	
	assert(player.health == initial_health - 20, "Damage should reduce health")
	
	player.queue_free()
