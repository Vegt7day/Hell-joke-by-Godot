创建通用的抛射体系统

我将为您创建一个通用的抛射体系统，用于学生的"鞋"和"墨"系统。这个系统将具有模块化设计，便于重用和扩展。

1. 创建通用抛射体基类

首先创建抛射体基类，提供通用的碰撞、移动、回收功能：
# projectiles/projectile_base.gd
extends Area2D
class_name ProjectileBase

# 信号
signal projectile_spawned
signal projectile_hit(target: Node, position: Vector2)
signal projectile_destroyed
signal projectile_recovered

# 导出属性
@export_category("基本属性")
@export var projectile_speed: float = 300.0
@export var projectile_damage: float = 10.0
@export var projectile_lifetime: float = 5.0
@export var gravity_enabled: bool = false
@export var gravity_strength: float = 98.0
@export var bounce_enabled: bool = false
@export var bounce_count: int = 3
@export var destroy_on_hit: bool = true
@export var is_recoverable: bool = true
@export var auto_recover_after_time: float = 0.0  # 0=不自动回收

# 引用
@onready var sprite: Sprite2D = $Sprite2D
@onready var collision_shape: CollisionShape2D = $CollisionShape2D
@onready var animation_player: AnimationPlayer = $AnimationPlayer
@onready var particle_system: CPUParticles2D = $Particles

# 内部属性
var velocity: Vector2 = Vector2.ZERO
var direction: Vector2 = Vector2.RIGHT
var is_active: bool = false
var owner_node: Node
var owner_team: String = ""
var hit_targets: Array[Node] = []
var current_bounce_count: int = 0
var lifetime_timer: float = 0.0
var recovery_timer: float = 0.0
var projectile_type: String = "generic"

func _ready():
	# 默认不活动
	set_active(false)
	
	# 连接信号
	area_entered.connect(_on_area_entered)
	body_entered.connect(_on_body_entered)
	
	print("抛射体基类初始化完成: %s" % name)

func _physics_process(delta: float):
	if not is_active:
		return
	
	# 更新生命周期计时器
	if projectile_lifetime > 0:
		lifetime_timer += delta
		if lifetime_timer >= projectile_lifetime:
			destroy()
			return
	
	# 更新自动回收计时器
	if auto_recover_after_time > 0:
		recovery_timer += delta
		if recovery_timer >= auto_recover_after_time:
			recover()
			return
	
	# 应用重力
	if gravity_enabled:
		velocity.y += gravity_strength * delta
	
	# 移动抛射体
	position += velocity * delta
	
	# 更新旋转（如果需要）
	if velocity.length() > 0:
		update_rotation()

func update_rotation():
	"""更新抛射体旋转"""
	if velocity.length() > 0:
		var angle = velocity.angle()
		rotation = angle

func setup_projectile(start_position: Vector2, move_direction: Vector2, owner: Node, team: String = "", type: String = "generic"):
	"""设置抛射体初始状态"""
	position = start_position
	direction = move_direction.normalized()
	velocity = direction * projectile_speed
	owner_node = owner
	owner_team = team
	projectile_type = type
	
	# 设置激活
	set_active(true)
	
	# 重置计时器
	lifetime_timer = 0.0
	recovery_timer = 0.0
	hit_targets.clear()
	current_bounce_count = 0
	
	# 发射信号
	projectile_spawned.emit()
	
	print("抛射体设置完成: 位置=%s, 方向=%s, 速度=%s" % [position, direction, velocity])

func set_active(active: bool):
	"""设置抛射体活动状态"""
	is_active = active
	
	# 控制碰撞检测
	collision_shape.disabled = not active
	
	# 控制可见性
	visible = active
	
	# 控制物理处理
	set_physics_process(active)
	
	print("抛射体活动状态: %s" % active)

func set_movement_type(move_type: String, params: Dictionary = {}):
	"""设置移动类型"""
	match move_type:
		"straight":
			# 直线运动
			gravity_enabled = false
			print("设置直线运动")
		
		"parabolic":
			# 抛物线运动
			gravity_enabled = true
			gravity_strength = params.get("gravity", 98.0)
			print("设置抛物线运动，重力: %f" % gravity_strength)
		
		"homing":
			# 追踪目标（需要目标节点）
			gravity_enabled = false
			# 这里可以添加追踪逻辑
			print("设置追踪运动")
		
		"sinusoidal":
			# 正弦波运动
			gravity_enabled = false
			# 这里可以添加正弦波逻辑
			print("设置正弦波运动")
		
		"boomerang":
			# 回旋镖运动
			gravity_enabled = false
			print("设置回旋镖运动")

func _on_area_entered(area: Area2D):
	"""与区域碰撞"""
	if not is_active:
		return
	
	# 检查是否是有效目标
	if is_valid_target(area):
		handle_collision(area, area.global_position)
	
	# 处理弹跳
	if bounce_enabled and current_bounce_count < bounce_count:
		handle_bounce(area)

func _on_body_entered(body: PhysicsBody2D):
	"""与物理体碰撞"""
	if not is_active:
		return
	
	# 检查是否是有效目标
	if is_valid_target(body):
		handle_collision(body, body.global_position)
	
	# 处理弹跳
	if bounce_enabled and current_bounce_count < bounce_count:
		handle_bounce(body)

func is_valid_target(target: Node) -> bool:
	"""检查是否是有效目标"""
	# 避免自碰撞
	if target == self or target == owner_node:
		return false
	
	# 避免重复击中同一目标
	if target in hit_targets:
		return false
	
	# 这里可以添加团队检测
	# 例如：if target.has_method("get_team") and target.get_team() == owner_team:
	#     return false
	
	return true

func handle_collision(target: Node, hit_position: Vector2):
	"""处理碰撞"""
	print("抛射体碰撞: 目标=%s, 位置=%s" % [target.name, hit_position])
	
	# 记录已击中的目标
	hit_targets.append(target)
	
	# 发射碰撞信号
	projectile_hit.emit(target, hit_position)
	
	# 应用伤害
	if target.has_method("take_damage"):
		target.take_damage(projectile_damage, projectile_type)
	
	# 如果碰撞后销毁
	if destroy_on_hit:
		destroy()
	
	# 播放碰撞效果
	play_hit_effect(hit_position)

func handle_bounce(collider: Node):
	"""处理弹跳"""
	current_bounce_count += 1
	
	# 获取碰撞法线
	var collision_point = get_collision_point(collider)
	var normal = get_collision_normal(collision_point)
	
	# 计算反射方向
	if normal != Vector2.ZERO:
		velocity = velocity.bounce(normal) * 0.8  # 减少能量
	
	print("抛射体弹跳: 剩余次数=%d" % (bounce_count - current_bounce_count))
	
	# 播放弹跳效果
	play_bounce_effect()

func get_collision_point(collider: Node) -> Vector2:
	"""获取碰撞点"""
	# 简单的碰撞点估计
	return global_position

func get_collision_normal(point: Vector2) -> Vector2:
	"""获取碰撞法线"""
	# 这里可以实现更精确的法线计算
	# 暂时使用简化版本
	return Vector2.UP

func play_hit_effect(hit_position: Vector2):
	"""播放击中效果"""
	if animation_player and animation_player.has_animation("hit"):
		animation_player.play("hit")
	
	# 触发粒子系统
	if particle_system:
		particle_system.emitting = true
		particle_system.global_position = hit_position

func play_bounce_effect():
	"""播放弹跳效果"""
	if animation_player and animation_player.has_animation("bounce"):
		animation_player.play("bounce")

func destroy():
	"""销毁抛射体"""
	print("销毁抛射体")
	
	# 播放销毁动画
	play_destroy_effect()
	
	# 发射销毁信号
	projectile_destroyed.emit()
	
	# 延迟后移除
	await get_tree().create_timer(0.5).timeout
	queue_free()

func recover():
	"""回收抛射体"""
	if not is_recoverable:
		destroy()
		return
	
	print("回收抛射体")
	
	# 播放回收动画
	play_recover_effect()
	
	# 发射回收信号
	projectile_recovered.emit()
	
	# 通知所有者
	if owner_node and owner_node.has_method("on_projectile_recovered"):
		owner_node.on_projectile_recovered(self)
	
	# 延迟后移除
	await get_tree().create_timer(0.3).timeout
	queue_free()

func play_destroy_effect():
	"""播放销毁效果"""
	if animation_player and animation_player.has_animation("destroy"):
		animation_player.play("destroy")

func play_recover_effect():
	"""播放回收效果"""
	if animation_player and animation_player.has_animation("recover"):
		animation_player.play("recover")

基于通用抛射体创建鞋子的具体实现：
# projectiles/shoe_projectile.gd
extends "res://projectiles/projectile_base.gd"
class_name ShoeProjectile

# 鞋特有属性
var shoe_owner: String = ""  # "left_shoe" 或 "right_shoe"
var is_slippery: bool = false
var trail_effect: bool = true
var trail_color: Color = Color(0.5, 0.3, 0.1)  # 棕色
var is_stuck: bool = false
var stuck_position: Vector2 = Vector2.ZERO
var can_be_picked_up: bool = true

func _ready():
	super._ready()
	
	# 设置鞋默认属性
	projectile_type = "shoe"
	bounce_enabled = true
	bounce_count = 2
	is_recoverable = true
	
	print("鞋抛射体初始化: %s" % name)

func setup_shoe(start_position: Vector2, direction: Vector2, owner: Node, shoe_name: String, params: Dictionary = {}):
	"""设置鞋抛射体"""
	# 调用父类设置
	setup_projectile(start_position, direction, owner, "", "shoe")
	
	# 设置鞋特有属性
	shoe_owner = shoe_name
	
	# 应用参数
	apply_shoe_params(params)
	
	# 设置初始速度
	var throw_power = params.get("throw_power", 1.0)
	velocity = direction * projectile_speed * throw_power
	
	# 应用旋转
	if params.get("rotate", true):
		var rotation_speed = params.get("rotation_speed", 10.0)
		rotation_speed = rotation_speed * (1.0 if direction.x >= 0 else -1.0)
		
		# 创建旋转Tween
		var tween = create_tween()
		tween.set_loops()
		tween.tween_property(self, "rotation", rotation + PI * 2, rotation_speed)
	
	print("鞋抛射体设置: 所有者=%s, 速度=%s" % [shoe_name, velocity])

func apply_shoe_params(params: Dictionary):
	"""应用鞋参数"""
	if params.has("slippery"):
		is_slippery = params.slippery
	if params.has("trail_effect"):
		trail_effect = params.trail_effect
	if params.has("trail_color"):
		trail_color = params.trail_color
	if params.has("damage"):
		projectile_damage = params.damage
	if params.has("bounce_count"):
		bounce_count = params.bounce_count
		bounce_enabled = bounce_count > 0

func handle_collision(target: Node, hit_position: Vector2):
	"""处理鞋碰撞"""
	print("鞋碰撞: 目标=%s, 鞋=%s" % [target.name, shoe_owner])
	
	# 检查是否卡在墙上
	if target is TileMap or target is StaticBody2D:
		if not is_stuck:
			stuck_position = hit_position
			is_stuck = true
			stick_to_surface(target, hit_position)
	
	# 调用父类处理
	super.handle_collision(target, hit_position)
	
	# 如果是地面，创建滑动效果
	if target is TileMap and is_slippery:
		create_slippery_area(hit_position)

func stick_to_surface(surface: Node, position: Vector2):
	"""粘在表面上"""
	print("鞋粘在表面上")
	
	# 停止运动
	set_active(false)
	
	# 固定在位置
	global_position = position
	
	# 播放粘住动画
	if animation_player and animation_player.has_animation("stick"):
		animation_player.play("stick")
	
	# 设置可拾取
	if can_be_picked_up:
		enable_pickup()

func enable_pickup():
	"""启用拾取"""
	print("鞋可拾取")
	
	# 添加拾取区域
	var pickup_area = Area2D.new()
	pickup_area.collision_layer = 4  # 拾取层
	pickup_area.collision_mask = 8   # 玩家层
	
	var pickup_shape = CollisionShape2D.new()
	pickup_shape.shape = CircleShape2D.new()
	pickup_shape.shape.radius = 20
	pickup_area.add_child(pickup_shape)
	
	pickup_area.area_entered.connect(_on_pickup_area_entered)
	add_child(pickup_area)

func _on_pickup_area_entered(area: Area2D):
	"""拾取区域进入"""
	print("拾取区域进入: %s" % area.name)
	
	# 检查是否是玩家
	var player = area.get_parent()
	if player and player.has_method("pickup_shoe"):
		if player.pickup_shoe(shoe_owner, self):
			# 被拾取，销毁
			destroy()

func create_slippery_area(position: Vector2):
	"""创建滑溜区域"""
	print("创建滑溜区域")
	
	# 创建滑溜区域
	var slippery_area = Area2D.new()
	slippery_area.collision_layer = 2
	slippery_area.collision_mask = 1
	
	var shape = CollisionShape2D.new()
	shape.shape = CircleShape2D.new()
	shape.shape.radius = 50
	slippery_area.add_child(shape)
	
	# 设置滑溜效果脚本
	slippery_area.body_entered.connect(
		func(body):
			if body.has_method("apply_slippery"):
				body.apply_slippery(3.0)  # 滑溜3秒
	)
	
	add_child(slippery_area)


4. 创建墨弹抛射体

# projectiles/ink_projectile.gd
extends "res://projectiles/projectile_base.gd"
class_name InkProjectile

# 墨弹特有属性
var ink_color: Color = Color(0.1, 0.1, 0.3, 0.8)  # 深蓝色
var ink_splash_size: float = 50.0
var is_sticky: bool = true
var stick_duration: float = 3.0
var ink_consumption: float = 8.0
var can_splash: bool = true
var splash_damage: float = 5.0
var splash_radius: float = 100.0
var leaves_trail: bool = true
var trail_duration: float = 2.0

func _ready():
	super._ready()
	
	# 设置墨弹默认属性
	projectile_type = "ink"
	destroy_on_hit = true
	gravity_enabled = false
	is_recoverable = false
	
	print("墨弹抛射体初始化: %s" % name)

func setup_ink(start_position: Vector2, direction: Vector2, owner: Node, params: Dictionary = {}):
	"""设置墨弹"""
	# 调用父类设置
	setup_projectile(start_position, direction, owner, "", "ink")
	
	# 应用参数
	apply_ink_params(params)
	
	# 设置颜色
	if sprite:
		sprite.modulate = ink_color
	
	# 设置粒子颜色
	if particle_system:
		particle_system.color = ink_color
	
	# 设置初始速度
	var speed_multiplier = params.get("speed_multiplier", 1.0)
	velocity = direction * projectile_speed * speed_multiplier
	
	print("墨弹设置: 颜色=%s, 速度=%s" % [ink_color, velocity])

func apply_ink_params(params: Dictionary):
	"""应用墨弹参数"""
	if params.has("ink_color"):
		ink_color = params.ink_color
	if params.has("splash_size"):
		ink_splash_size = params.splash_size
	if params.has("is_sticky"):
		is_sticky = params.is_sticky
	if params.has("stick_duration"):
		stick_duration = params.stick_duration
	if params.has("ink_consumption"):
		ink_consumption = params.ink_consumption
	if params.has("can_splash"):
		can_splash = params.can_splash
	if params.has("splash_damage"):
		splash_damage = params.splash_damage
	if params.has("splash_radius"):
		splash_radius = params.splash_radius
	if params.has("leaves_trail"):
		leaves_trail = params.leaves_trail
	if params.has("trail_duration"):
		trail_duration = params.trail_duration

func handle_collision(target: Node, hit_position: Vector2):
	"""处理墨弹碰撞"""
	print("墨弹碰撞: 目标=%s" % target.name)
	
	# 调用父类处理
	super.handle_collision(target, hit_position)
	
	# 创建墨水飞溅
	if can_splash:
		create_ink_splash(hit_position)
	
	# 如果是可粘附的表面
	if is_sticky and (target is TileMap or target is StaticBody2D):
		stick_ink(hit_position)
	
	# 销毁墨弹
	destroy()

func create_ink_splash(position: Vector2):
	"""创建墨水飞溅"""
	print("创建墨水飞溅")
	
	# 创建飞溅区域
	var splash_area = Area2D.new()
	splash_area.collision_layer = 2
	splash_area.collision_mask = 1
	
	var shape = CollisionShape2D.new()
	shape.shape = CircleShape2D.new()
	shape.shape.radius = splash_radius
	splash_area.add_child(shape)
	
	# 创建飞溅效果
	var splash_sprite = Sprite2D.new()
	splash_sprite.texture = preload("res://assets/graphics/effects/ink_splash.png")
	splash_sprite.modulate = ink_color
	splash_sprite.scale = Vector2.ONE * (splash_radius / 50.0)
	splash_area.add_child(splash_sprite)
	
	# 添加飞溅逻辑
	splash_area.area_entered.connect(
		func(area):
			var target = area.get_parent()
			if target and target.has_method("take_damage"):
				target.take_damage(splash_damage, "ink_splash")
				apply_ink_debuff(target)
	)
	
	splash_area.body_entered.connect(
		func(body):
			if body.has_method("take_damage"):
				body.take_damage(splash_damage, "ink_splash")
				apply_ink_debuff(body)
	)
	
	# 添加到场景
	get_parent().add_child(splash_area)
	
	# 设置自动销毁
	var timer = splash_area.get_tree().create_timer(trail_duration)
	timer.timeout.connect(
		func():
			splash_area.queue_free()
	)

def apply_ink_debuff(target: Node):
	"""应用墨水减益效果"""
	if target.has_method("apply_debuff"):
		target.apply_debuff("ink_slow", {
			"duration": 2.0,
			"slow_amount": 0.5  # 减速50%
		})

def stick_ink(position: Vector2):
	"""粘附墨水"""
	print("墨水粘附在表面")
	
	# 创建粘附的墨迹
	var ink_stain = Sprite2D.new()
	ink_stain.texture = preload("res://assets/graphics/effects/ink_stain.png")
	ink_stain.modulate = ink_color
	ink_stain.global_position = position
	ink_stain.z_index = -1  # 在背景层
	
	# 添加到场景
	get_parent().add_child(ink_stain)
	
	# 设置自动消失
	var timer = ink_stain.get_tree().create_timer(stick_duration)
	timer.timeout.connect(
		func():
			# 淡出效果
			var fade_tween = create_tween()
			fade_tween.tween_property(ink_stain, "modulate:a", 0.0, 0.5)
			fade_tween.tween_callback(ink_stain.queue_free)
	)


5. 创建抛射体管理器

# projectiles/projectile_manager.gd
extends Node
class_name ProjectileManager

static var instance: ProjectileManager

# 预制体引用
var projectile_prefabs: Dictionary = {
	"shoe": "res://projectiles/shoe_projectile.tscn",
	"ink": "res://projectiles/ink_projectile.tscn"
}

# 对象池
var projectile_pools: Dictionary = {}

# 活动中的抛射体
var active_projectiles: Array[ProjectileBase] = []

func _init():
	if instance == null:
		instance = self
	else:
		queue_free()

func _ready():
	print("抛射体管理器初始化")
	
	# 预加载预制体
	preload_projectiles()

func preload_projectiles():
	"""预加载抛射体预制体"""
	for type, path in projectile_prefabs.items():
		if ResourceLoader.exists(path):
			var prefab = load(path)
			if prefab:
				# 初始化对象池
				projectile_pools[type] = []
				print("预加载抛射体: %s" % type)
			else:
				print("警告: 无法加载预制体: %s" % path)
		else:
			print("警告: 预制体路径不存在: %s" % path)

func spawn_projectile(projectile_type: String, position: Vector2, direction: Vector2, owner: Node, params: Dictionary = {}) -> ProjectileBase:
	"""生成抛射体"""
	print("生成抛射体: 类型=%s, 位置=%s" % [projectile_type, position])
	
	# 获取或创建抛射体实例
	var projectile = get_projectile_from_pool(projectile_type)
	if not projectile:
		return null
	
	# 添加到活动列表
	active_projectiles.append(projectile)
	
	# 添加到场景
	get_tree().current_scene.add_child(projectile)
	
	# 根据类型设置
	match projectile_type:
		"shoe":
			var shoe_name = params.get("shoe_name", "shoe")
			projectile.setup_shoe(position, direction, owner, shoe_name, params)
		"ink":
			projectile.setup_ink(position, direction, owner, params)
		_:
			projectile.setup_projectile(position, direction, owner, "", projectile_type)
	
	# 连接信号
	projectile.projectile_destroyed.connect(_on_projectile_destroyed.bind(projectile))
	projectile.projectile_recovered.connect(_on_projectile_recovered.bind(projectile))
	
	return projectile

func get_projectile_from_pool(projectile_type: String) -> ProjectileBase:
	"""从对象池获取抛射体"""
	# 检查对象池
	if projectile_type in projectile_pools and projectile_pools[projectile_type].size() > 0:
		print("从对象池获取抛射体: %s" % projectile_type)
		return projectile_pools[projectile_type].pop_back()
	
	# 创建新实例
	if projectile_type in projectile_prefabs:
		var prefab = load(projectile_prefabs[projectile_type])
		if prefab:
			var instance = prefab.instantiate()
			print("创建新抛射体: %s" % projectile_type)
			return instance
	
	print("错误: 无法创建抛射体: %s" % projectile_type)
	return null

func return_projectile_to_pool(projectile: ProjectileBase):
	"""将抛射体返回对象池"""
	var type = projectile.projectile_type
	
	# 初始化对象池
	if type not in projectile_pools:
		projectile_pools[type] = []
	
	# 重置抛射体状态
	projectile.set_active(false)
	projectile.visible = false
	
	# 添加到对象池
	projectile_pools[type].append(projectile)
	
	print("抛射体返回对象池: %s" % type)

func _on_projectile_destroyed(projectile: ProjectileBase):
	"""抛射体销毁"""
	print("抛射体销毁: %s" % projectile.name)
	
	# 从活动列表移除
	if projectile in active_projectiles:
		active_projectiles.erase(projectile)
	
	# 延迟后实际移除
	await get_tree().create_timer(0.5).timeout
	if is_instance_valid(projectile):
		projectile.queue_free()

func _on_projectile_recovered(projectile: ProjectileBase):
	"""抛射体回收"""
	print("抛射体回收: %s" % projectile.name)
	
	# 从活动列表移除
	if projectile in active_projectiles:
		active_projectiles.erase(projectile)
	
	# 返回对象池
	return_projectile_to_pool(projectile)

func clear_all_projectiles():
	"""清除所有抛射体"""
	print("清除所有抛射体")
	
	for projectile in active_projectiles.duplicate():
		if is_instance_valid(projectile):
			projectile.queue_free()
	
	active_projectiles.clear()
	
	# 清空对象池
	for type in projectile_pools.keys():
		for projectile in projectile_pools[type]:
			if is_instance_valid(projectile):
				projectile.queue_free()
		projectile_pools[type].clear()
